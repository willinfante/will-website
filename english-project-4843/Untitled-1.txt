    function changeDown(currIndex) {

        try {

        }

    function checkKey(e) {

        e = e || window.event;

        if (e.keyCode == '37') {
            // left arrow
            if (i > 1) {
                i--;
                changeDown(i);
            }
        }
        else if (e.keyCode == '39') {
            // right arrow
            if (i < 20) {
                i++;
                changeUp(i);

            }
        }
